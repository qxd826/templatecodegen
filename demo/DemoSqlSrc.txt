CREATE TABLE AO_PROJECT (
	ID                       BIGINT(14) NOT NULL AUTO_INCREMENT COMMENT 'ID#INPUT',
	GMT_CREATE               DATETIME NOT NULL COMMENT '创建时间#DATE', 
	GMT_MODIFIED             DATETIME NOT NULL COMMENT '修改时间#DATE', 
	CREATOR                  BIGINT(14) NOT NULL COMMENT '创建人#INPUT', 
	MODIFIER                 BIGINT (14) NOT NULL COMMENT '修改人#INPUT', 
	IS_DELETED               CHAR(1) NOT NULL DEFAULT 'N' COMMENT '是否删除#CHECKBOX',
	PROJECT_NAME             VARCHAR(128) NOT NULL COMMENT '项目名称#INPUT',
	PROJECT_TYPE             TINYINT UNSIGNED NOT NULL COMMENT '项目类型#DROPDOWN',
	START_DATE               DATETIME NOT NULL COMMENT '启动日期#DATE',
	DEPARTMENT               BIGINT(14) NOT NULL COMMENT '所属部门ID#DROPDOWN',
	PUBLICITY                CHAR(1) NOT NULL DEFAULT 'N' COMMENT '是否公开#CHECKBOX', 
	IS_IMPORTANT             CHAR(1) NOT NULL DEFAULT 'N' COMMENT '是否重要项目#CHECKBOX', 
	PROJECT_GOAL             VARCHAR(4000) COMMENT '项目目标#TEXTAREA', 
	PROJECT_SCOPE            VARCHAR(4000) COMMENT '项目范围#TEXTAREA',
	PROJECT_STATUS           TINYINT UNSIGNED NOT NULL COMMENT '状态#DROPDOWN',
	EXPECT_RELEASE_DATE      DATETIME COMMENT '期望上线日期#DATE',
	EXPECT_INITIATE_DATE     DATETIME COMMENT '期望立项日期#DATE',
	PRODUCT_MANAGER          BIGINT(14) COMMENT '产品主管#INPUT', 
	DEVELOP_MANAGER          BIGINT(14) COMMENT '开发主管#INPUT', 

	PRIMARY KEY (ID)
)
